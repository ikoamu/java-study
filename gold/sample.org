  String name;

  public Car(int cno, String name) {
    this.cno = cno;
    this.name = name;
  }

  public boolean equals(Object obj) {
    if (obj == null) { return false; }

    Car other = (Car) obj;

    if (this.name.equals(other.name)) {
      return true;
    } else {
      return false;
    }
  }
}
